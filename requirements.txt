# RAIM Scope — Python dependencies
#
# Recommended setup on Raspberry Pi (Debian/Trixie):
#   sudo apt install -y python3-pyqt5 python3-opencv python3-numpy sense-hat \
#                       python3-venv python3-pip
#   uv venv --system-site-packages .venv
#   .venv/bin/python -m app.main
#
# The venv inherits the apt-installed packages above, so no pip install
# is required on Pi. The pinned constraints below are fallbacks for non-Pi
# systems (Windows/macOS dev machines) where you can install via pip:
#   uv pip install -r requirements.txt
#
# Tested on:
#   - Raspberry Pi 5  / Debian 13 (trixie) / Python 3.13.5  (target)
#   - Windows 11      / openCV conda env   / Python 3.12.7  (dev)

PyQt5>=5.15,<6
opencv-python>=4.5
numpy>=1.20

# On Pi: install via apt (`sudo apt install sense-hat`), not pip.
